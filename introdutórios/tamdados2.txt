#include <iostream>
using namespace std;

int main(){
    cout << "Variável 'short' ocupa           : " << sizeof(short) << "byte(s)" << endl;
    cout << "Variável 'unsigned short' ocupa  : " << sizeof(unsigned short) << "byte(s)" << endl;
    cout << "Variável 'unsigned int' ocupa    : " << sizeof(unsigned int) << "byte(s)" << endl;
    cout << "Variável 'long' ocupa            : " << sizeof(long) << "byte(s)" << endl;
    cout << "Variável 'unsigned long' ocupa   : " << sizeof(unsigned long) << "byte(s)" << endl;
    
    return 0;
}
