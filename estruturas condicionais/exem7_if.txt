#include <iostream>
using namespace std;

int main() {
    float idade;
    
    cout << "Digite a sua idade: ";
    cin >> idade;

    //Condição de negação, ele troca o condição de verdadeiro ou falso
    if(!(idade >= 18 && idade <= 65)) {
        cout << "Voto facultativo";
    }
    
    else if(idade >= 18) {
        cout << "Voto obrigatório";
    } 
    else {
        cout << "Você não tem idade para vota!";
    }
   
    return 0;
    
}
